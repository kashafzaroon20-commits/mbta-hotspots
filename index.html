import pandas as pd import numpy as np from sklearn.cluster import KMeans from sklearn.preprocessing import StandardScaler import matplotlib.pyplot as plt # ── 1. LOAD FILES ───────────────────────────────────────────── df = pd.read_csv("all_seasons_clean.csv") stops = pd.read_csv("stops.txt") df["stop_id"] = df["stop_id"].astype(str) stops["stop_id"] = stops["stop_id"].astype(str) print(f"Loaded: {len(df):,} rows") print(f"Columns: {df.columns.tolist()}") # ── 2. JOIN LAT/LON ─────────────────────────────────────────── df = df.merge( stops[["stop_id", "stop_lat", "stop_lon"]], on="stop_id", how="left" ) df = df.dropna(subset=["stop_lat", "stop_lon", "score"]) print(f"Stops with coordinates: {len(df):,}") # ── 3. CONVERT MEAN DELAY TO SECONDS ────────────────────────── df["mean_delay_sec"] = df["mean_delay_min"] * 60 # ── 4. SCALE FEATURES ───────────────────────────────────────── features = df[["stop_lat", "stop_lon", "mean_delay_sec", "pct_late", "score"]] scaler = StandardScaler() f_scaled = scaler.fit_transform(features) # ── 5. ELBOW METHOD ─────────────────────────────────────────── print("\nRunning elbow method...") inertias = [] k_range = range(2, 15) for k in k_range: km = KMeans(n_clusters=k, random_state=42, n_init=10) km.fit(f_scaled) inertias.append(km.inertia_) print(f"k={k}: inertia={km.inertia_:.1f}") # Plot and save elbow chart plt.figure(figsize=(8, 4)) plt.plot(list(k_range), inertias, marker='o', color='steelblue') plt.xlabel("Number of clusters (k)") plt.ylabel("Inertia") plt.title("Elbow method — all seasons combined") plt.xticks(list(k_range)) plt.tight_layout() plt.savefig("elbow_chart.png", dpi=150) plt.show() print("\nElbow chart saved as elbow_chart.png") # ── 6. SET K BASED ON ELBOW ─────────────────────────────────── # Change this number based on the elbow chart output K = 6 # ── 7. K-MEANS ──────────────────────────────────────────────── print(f"\nRunning K-means with k={K}...") km = KMeans(n_clusters=K, random_state=42, n_init=10) df["cluster"] = km.fit_predict(f_scaled) # Label by severity cluster_means = df.groupby("cluster")["score"].mean().sort_values(ascending=False) labels = ["Severe", "High", "Moderate-High", "Moderate", "Low-Moderate", "Low"][:K] label_map = {c: l for c, l in zip(cluster_means.index, labels)} df["cluster_label"] = df["cluster"].map(label_map) print("\nCluster summary:") print(df.groupby("cluster_label").agg( mean_delay_min=("mean_delay_min", "mean"), stop_count=("stop_id", "count"), pct_late_avg=("pct_late", "mean"), avg_income=("median_household_income", "mean") ).round(2)) # ── 8. SAVE CSV ─────────────────────────────────────────────── df.to_csv("all_seasons_clustered.csv", index=False) print("\nSaved: all_seasons_clustered.csv") # ── 9.
